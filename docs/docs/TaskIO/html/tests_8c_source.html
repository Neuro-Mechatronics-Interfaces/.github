167static void testCreateAndWriteCursorLog(const char* subject, HeaderData* headerData, LoggerConfig* config, ThreadSafeCursorArgs* args) {
168 pthread_t clock_thread, producer_thread, consumer_thread;
169
170 pthread_attr_t attr;
171 struct sched_param param;
172 args->log = createCursorLog(subject, 1, headerData, config);
173 if (!args->log) {
174 fprintf(stderr, "Error: Failed to create cursor log\n");
175 return;
176 }
177 pthread_attr_init(&attr);
178 pthread_attr_setschedpolicy(&attr, SCHED_FIFO);
179 param.sched_priority = sched_get_priority_max(SCHED_FIFO);
180 pthread_attr_setschedparam(&attr, &param);
181
182 // Starts cursor producer thread (simulates cursor position)
183 startCursorThread(args, cursor_producer_thread_func, &producer_thread, &args->initx, NULL, NULL);
184
185 // Starts cursor consumer thread (saves cursor logs)
186 startCursorThread(args, cursor_consumer_thread_func, &consumer_thread, &args->initx, NULL, NULL);
187
188 // Sleep for 50-ms ensuring we acquire mutex lock on Cursor Clock thread:
189 Sleep(50);
190
191 // Starts cursor clock (issues "tick" trigger periodically)
192 startCursorThread(args, cursor_clock_thread_func, &clock_thread, NULL, NULL, &attr);
193
194 pthread_mutex_lock(&args->mutex);
195 printf("Threads created and running...\n");
196 pthread_mutex_unlock(&args->mutex);
197 // Wait for the consumer thread to finish processing all data
198 pthread_join(consumer_thread, NULL);
199 pthread_mutex_lock(&args->mutex);
200 printf("Joined consumer thread.\n");
201 pthread_mutex_unlock(&args->mutex);
202 pthread_join(clock_thread, NULL);
203 pthread_mutex_lock(&args->mutex);
204 printf("Joined clock thread.\n");
205 pthread_mutex_unlock(&args->mutex);
206 pthread_join(producer_thread, NULL);
207 pthread_mutex_lock(&args->mutex);
208 printf("Joined producer thread.\n");
209 pthread_mutex_unlock(&args->mutex);
210}
