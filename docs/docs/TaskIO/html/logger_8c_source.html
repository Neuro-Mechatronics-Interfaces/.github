49void *cursor_clock_thread_func(void *arg) {
50 ThreadSafeCursorArgs* args = (ThreadSafeCursorArgs*)arg;
51 LARGE_INTEGER main_start, tick_start, tick_end;
52 QueryPerformanceCounter(&main_start);
53 double elapsed_time;
54 tick_start = main_start;
55 do {
56 // Get current time
57 pthread_mutex_lock(&args->mutex);
58 // Calculate elapsed time in milliseconds
59 QueryPerformanceCounter(&tick_end);
60 elapsed_time = getElapsedTime(tick_start.QuadPart, tick_end.QuadPart);
61 if (elapsed_time >= args->sample_period) { // If longer than 1 ms:
62 // Issue trigger by signaling producer thread
63 tick_start = tick_end;
64 args->cursorData->ts = getElapsedTime(main_start.QuadPart, tick_end.QuadPart);
65 // printf("Tick.\n");
66 pthread_cond_signal(&args->tick);
67 pthread_cond_wait(&args->tick, &args->mutex);
68 }
69 pthread_mutex_unlock(&args->mutex);
70 } while (args->running);
71}

79void *cursor_producer_thread_func(void *arg) {
80 ThreadSafeCursorArgs* args = (ThreadSafeCursorArgs*)arg;
81 // Simulate a 200-Hz logger for 5 seconds
82 size_t seg = 1;
83 const size_t numControlPoints = NUM_CURSOR_CONTROL_POINTS;
84 double tControlPoints[numControlPoints];
85 double xControlPoints[numControlPoints];
86 double yControlPoints[numControlPoints];
87 double t = 0.0;
88 double ttotal = 5.0;
89
90 // Allocate memory
91 uint16_t* x = malloc(sizeof(uint16_t));
92 uint16_t* y = malloc(sizeof(uint16_t));
93
94 // Generate random control points
95 generateRandomControlTimes(tControlPoints, numControlPoints);
96 generateRandomControlPoints(xControlPoints, CURSOR_X_MAX, numControlPoints);
97 generateRandomControlPoints(yControlPoints, CURSOR_Y_MAX, numControlPoints);
98 pthread_cond_signal(&args->init); // indicate initialization is done
99 do {
100 pthread_mutex_lock(&args->mutex);
101 // printf("Producer has lock awaiting tick.\n");
102 pthread_cond_wait(&args->tick, &args->mutex);
103 t = args->cursorData->ts / ttotal;
104
105 // Increment sample and get system timestamp
106 cubicSpline2DInterpolant(t, tControlPoints, xControlPoints, yControlPoints, &seg, numControlPoints, x, y);
107 args->cursorData->x = *x;
108 args->cursorData->y = *y;
109
110 pthread_cond_signal(&args->update); // Signal consumer thread
111 pthread_cond_signal(&args->tick);
112 pthread_mutex_unlock(&args->mutex);
113 } while (t <= 1.0);
114 pthread_mutex_lock(&args->mutex);
115 args->running = 0;
116 pthread_cond_signal(&args->update);
117 pthread_cond_signal(&args->tick);
118 pthread_mutex_unlock(&args->mutex);
119 free(x);
120 free(y);
121}

249int readConfig(LoggerConfig* config) {
250 // Get the path of the executable
251 char exePath[MAX_PATH];
252 GetModuleFileName(NULL, exePath, MAX_PATH);
253
254 // Remove the executable name from the path
255 char* lastSlash = strrchr(exePath, '\\');
256 if (lastSlash != NULL) {
257 *(lastSlash + 1) = '\0';
258 }
259
260 // Construct the full path to config.yaml
261 char configPath[MAX_PATH];
262 snprintf(configPath, sizeof(configPath), "%sconfig.yaml", exePath);
263
264 // Open config.yaml for reading
265 FILE* configFile = fopen(configPath, "r");
266 if (!configFile) {
267 fprintf(stderr, "Error: Unable to open config.yaml\n");
268 return 0;
269 }
270
271 char line[256]; // Adjust the size as needed
272
273 // Read each line from the config file
274 while (fgets(line, sizeof(line), configFile)) {
275 // Ignore empty lines and comments
276 if (line[0] == '\n' || line[0] == '#') {
277 continue;
278 }
279
280 // Tokenize the line to extract parameter and value
281 char parameter[128]; // Adjust the size as needed
282 char value[128]; // Adjust the size as needed
283
284 if (sscanf(line, "%127[^:]: \"%127[^\"]\"", parameter, value) == 2) {
285 // Check the parameter and set the corresponding value in the config struct
286 if (strcmp(parameter, "LogSaveFolder") == 0) {
287 strncpy(config->logSaveFolder, value, sizeof(config->logSaveFolder) - 1);
288 config->logSaveFolder[sizeof(config->logSaveFolder) - 1] = '\0'; // Ensure null-termination
289 } else if (strcmp(parameter, "LogReadFolder") == 0) {
290 strncpy(config->logReadFolder, value, sizeof(config->logReadFolder) - 1);
291 config->logReadFolder[sizeof(config->logReadFolder) - 1] = '\0'; // Ensure null-termination
292 }
293 // Add more checks for other parameters if needed
294 }
295 }
296
297 // Check if logSaveFolder ends with the file separator
298 size_t lenSave = strlen(config->logSaveFolder);
299 if (lenSave > 0 && config->logSaveFolder[lenSave - 1] != '\\') {
300 strcat(config->logSaveFolder, "\\");
301 }
302
303 // Check if logReadFolder ends with the file separator
304 size_t lenRead = strlen(config->logReadFolder);
305 if (lenRead > 0 && config->logReadFolder[lenRead - 1] != '\\') {
306 strcat(config->logReadFolder, "\\");
307 }
308
309 // Close the file
310 int res = fclose(configFile);
311 if (res==0) {

314 perror("Error closing config file");

327static int createFolderIfNotExists(const char* folderPath) {
328 if (CreateDirectory(folderPath, NULL)) {
329 return 1; // Success
330 }
331
332 // Check if the folder already exists
333 if (GetLastError() == ERROR_ALREADY_EXISTS) {
334 return 1; // Success
335 }
336
337 // Check if the error is related to non-existing parent directories
338 DWORD error = GetLastError();
339 if (error == ERROR_PATH_NOT_FOUND || error == ERROR_FILE_NOT_FOUND) {
340 // Attempt to create parent directories
341 size_t len = strlen(folderPath);
342 char* parentPath = malloc(len + 1);
343 strncpy(parentPath, folderPath, len);
344 parentPath[len] = '\0';
345
346 // Find the last backslash to get the parent path
347 char* lastBackslash = strrchr(parentPath, '\\');
348 if (lastBackslash != NULL) {
349 *lastBackslash = '\0';
350 int parentResult = createFolderIfNotExists(parentPath);
351
352 // Free allocated memory
353 free(parentPath);
354
355 // Return success or failure based on parent directory creation
356 return parentResult && CreateDirectory(folderPath, NULL);
357 }
358
359 // Free allocated memory
360 free(parentPath);
361 }
362
363 fprintf(stderr, "Error: Unable to create folder (%s)\n", folderPath);
364 return 1; // Success
365}

375static void replaceBackslashes(char* filePath, char* filePathFS) {
376 size_t length = strlen(filePath);
377
378 for (size_t i = 0; i < length; ++i) {
379 if (filePath[i] == '\\') {
380 filePathFS[i] = '/'; // Replace backslash with forward slash
381 } else {
382 filePathFS[i] = filePath[i]; // Copy other characters unchanged
383 }
384 }
385
386 filePathFS[length] = '\0'; // Null-terminate the new string
387}

432static FILE* openLogFileToWrite(const char* subject, const char* tag, uint16_t index, size_t chunkSize, HeaderData* headerData, LoggerConfig* config) {
433 TCHAR logFilePath[MAX_PATH];
434 TCHAR logFilePathFS[MAX_PATH];
435 FILE* logFile;
436
437 // Read the configuration from config.yaml
438 if (!readConfig(config)) {
439 return NULL; // Failed to open config.yaml (stderr handled in readConfig)
440 }
441
442 // Create the folder if it doesn't exist
443 if (!createFolderIfNotExists(config->logSaveFolder)) {
444 return NULL; // Failed to create the folder (stderr in createFolderIfNotExists)
445 }
446
447 // Get the current date and time
448 time_t t = time(NULL);
449 struct tm now;
450 localtime_s(&now, &t);
451
452 // Try to open the log file with an incremented index if it already exists
453 do {
454 // Construct the full path to the log file with current date and index
455 snprintf(logFilePath, sizeof(logFilePath), TEXT("%s%s_%04d_%02d_%02d_%s_%d.log"),
456 config->logSaveFolder, subject, now.tm_year + 1900, now.tm_mon + 1, now.tm_mday, tag, index);
457 replaceBackslashes(logFilePath,logFilePathFS);
458 logFile = fopen(logFilePathFS, "rb");
459
460 // Check if the file already exists
461 if (logFile) {
462 fclose(logFile);
463 index++;
464 }
465 } while (logFile);
466
467 // Open the log file for writing
468 wprintf(L"Combined Path=<%s>\n", logFilePathFS);
469 logFile = fopen(TEXT(logFilePathFS), "wb");
470 if (!logFile) {
471 perror("Error opening log file");
472 fprintf(stderr, "Error: Unable to open log file (%s) for writing\n", logFilePathFS);
473 return NULL;
474 }
475
476 // Write a header with the tag and version information
477 if (addFileHeader(logFile, tag, index, chunkSize, headerData)) {
478 return logFile; // Could not write header information (stderr handled in addFileHeader)
479 } else {
480 return NULL;
481 }
482}

494static int addFileHeader(FILE* logFile, const char* tag, uint16_t index, size_t chunkSize, HeaderData* headerData) {
495 snprintf(headerData->logType, sizeof(headerData->logType), "%sLog", tag);
496 strcpy(headerData->logVersion, LOGGER_VERSION);
497 headerData->logIndex = index;
498 headerData->chunkSize = chunkSize;
499 headerData->timestamp = getCurrentTimestamp();
500 printHeaderData(headerData);
501 char timeBuffer[30];
502 formatTimestampWithMillis(*(FILETIME*)&headerData->timestamp, timeBuffer, 30);
503 if (fprintf(logFile, "Log Start: %s\n", timeBuffer) < 0 ||
504 fprintf(logFile, "Log Type: %s\n", headerData->logType) < 0 ||
505 fprintf(logFile, "Log Version: %s\n", headerData->logVersion) < 0 ||
506 fprintf(logFile, "Log Index: %d\n", headerData->logIndex) < 0 ||
507 fprintf(logFile, "ChunkSize: %d\n\n", headerData->chunkSize) < 0) {
508 if (!(fclose(logFile)==0)) {
509 perror("Error closing log file after failed header write");
510 }
511 fprintf(stderr, "Error: Failed to write header to log file\n");
512 return 0; // Failure
513 }
514 return 1; // Success
515}
